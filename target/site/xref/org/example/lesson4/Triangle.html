3   public class Triangle {
4       public double calcArea(int a, int b, int c) throws NegativeTriangleSideException, NonExistentTriangleException, ZeroTriangleSideException {
5           if ((a < 0) || (b < 0) || (c < 0)) {
6               throw new NegativeTriangleSideException();
7           }
8           if ((a == 0) || (b == 0) || (c == 0)) {
9               throw new ZeroTriangleSideException();
10          }
11          if ((a + b <= c) || (b + c <= a) || (c + a <= b)) {
12              throw new NonExistentTriangleException();
13          }
14          double p = (a + b + c) * 0.5;
15          return Math.sqrt(p * (p - a) * (p - b) * (p - c));
16      }
17  }
