20          WebDriver driver = new ChromeDriver(options);
21          driver.manage().timeouts().implicitlyWait(3, TimeUnit.SECONDS);
22  
23          driver.get("https://www.livejournal.com");
24          WebElement loginButton = driver.findElement(By.cssSelector(".s-header-item__link--login"));
25          loginButton.click();

27          loginInput.click();
28          loginInput.sendKeys("gb_web_ui");
29          WebElement passwordInput = driver.findElement(By.id("lj_loginwidget_password"));
30          passwordInput.click();
31          passwordInput.sendKeys("GB12345webui");
32          WebElement loginActionButton = driver.findElement(By.name("action:login"));
33          loginActionButton.click();
34  
35          WebElement menu = driver.findElement(By.cssSelector(".s-header-item.s-nav-item__name"));
36          Actions actions = new Actions(driver);
37          actions.moveToElement(menu).perform();
38          WebElement settingsMenuButton = driver.findElement(By.cssSelector(".s-header-sub-list-item__link--settings"));
39          settingsMenuButton.click();
40  
41          WebElement displaySettingsTab = driver.findElement(By.cssSelector(".b-menu-item:nth-child(3) > .b-menu-item-link"));
42          displaySettingsTab.click();
43          Select languageDropdownMenu = new Select(driver.findElement(By.id("LJ__Setting__Language_lang")));
44          languageDropdownMenu.selectByValue("en_LJ");
45          WebElement saveChangesButton = driver.findElement(By.cssSelector("#settings_save > .b-flatbutton"));
