11  import java.util.concurrent.TimeUnit;
12  
13  public class LoginLogoutLiveJournalTest {
14      public static void main(String[] args) {
15          WebDriverManager.chromedriver().setup();
16          ChromeOptions options = new ChromeOptions();
17          options.addArguments("--incognito");
18          options.addArguments("start-maximized");
19          WebDriver driver = new ChromeDriver(options);
20          driver.manage().timeouts().implicitlyWait(3, TimeUnit.SECONDS);
21  
22          driver.get("https://www.livejournal.com");
23          WebElement loginButton = driver.findElement(By.cssSelector(".s-header-item__link--login"));
24          loginButton.click();
25          WebElement loginInput = driver.findElement(By.id("user"));
26          loginInput.click();
27          loginInput.sendKeys("gb_web_ui");
28          WebElement passwordInput = driver.findElement(By.id("lj_loginwidget_password"));
29          passwordInput.click();
30          passwordInput.sendKeys("GB12345webui");
31          WebElement loginActionButton = driver.findElement(By.name("action:login"));
32          loginActionButton.click();
33          WebElement menu = driver.findElement(By.cssSelector(".s-nav-item__name"));
34          Actions actions = new Actions(driver);
35          actions.moveToElement(menu).perform();
36          WebElement logoutMenuButton = driver.findElement(By.cssSelector(".s-header-sub-list-item__link--logout"));
37          logoutMenuButton.click();
38          driver.close();
39      }
40  }
